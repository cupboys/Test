### 发送数据
```sh
#发送消息的策略；1：单独发送人脸数据；2：单独发送人体数据；3：人脸人体数据一起发送
#发送单条数据时分开发送即可，策略1，或#
sendStrategy=3

#批量发送与手动发送的开关;发送单条数据时，batchCount必须配置为1#
sendOneMsg = false

#手动发送单条数据的时候，数据json信息存放位置 truedata/oneFace#
faceMsgJsonInfoAddr=truedata/oneFace.txt
humanMsgJsonInfoAddr=truedata/oneHuman.txt

#存在本地的人脸数据量
total_num_face=1000
#存在本地的人体数据量
total_num_human=1000
#发送一批数据的数量（或小于total_num_face/total_num_human或是total_num_face/total_num_human的整数倍）
batchCount=4000

#开始时间（闭区间）
starttime=2018-05-15

#结束时间（闭区间）
endtime=2018-05-15

#每批次数据发送间隔时间/ms，批次越多间隔时间越长


#手动设置的sceneImgUrl#
sceneImgUrl=PFSB:/vdtimg/bimg/data/20170815/19_0/709f6bf8c3cfb00b2918c20f85aa3535

#手动设置的traitImgUrl#
traitImgUrl=PFSB:/vdtimg/bimg/data/20170815/19_0/bf303313504e16ab8f8aca7877840df8

#sourceType#
sourceType=1

#手动设置的traitLocation#
traitLocation=100.100.100.100

#自动设置sourceID的开关#
autoSetSourceID = false

#手动设置的sourceID#
sourceId = 1111111,2222222,33333333,4444444,555555,66666,77777,888888,999999

#读取的原始数据文件名称地址#
sourceFaceDataAddr=facedata/
sourceHumanDataAddr=humandata/

#执行的sql语句，主要是获取本地数据库的表的sourceID#
sqlString=SELECT id FROM proxy_camera

#kafka相关配置，目前仅支持单机版kafka
#人脸的topicname
service.receive.topic.name_face=face_features
#人体的topicname
service.receive.topic.name_human=human_features
bootstrap.servers=node122:9092

```
